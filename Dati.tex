\newcommand{\prospettoSettimanale}{
    \begin{itemize}
        \item \textbf{Prima Settimana (40 ore) --- Formazione: Onboarding e introduzione al progetto}
        \begin{itemize}
            \item Incontro con le persone coinvolte nel progetto;
            \item Verifica credenziali e strumenti di lavoro assegnati;
            \item Presa visione dell’infrastruttura esistente;
            \item Introduzione al progetto e agli obiettivi dello stage;
        \end{itemize}
        \item \textbf{Seconda Settimana (40 ore) --- Formazione: Studio delle tecnologie}
        \begin{itemize}
            \item Studio della piattaforma Elastic (Elasticsearch e Kibana);
            \item Studio della piattaforma AppDynamics e delle API di integrazione;
            \item Approfondimento dei modelli LLM disponibili (OpenAI, Anthropic, Microsoft);
        \end{itemize}
        \item \textbf{Terza Settimana (40 ore) --- Analisi e progettazione}
        \begin{itemize}
            \item Analisi delle specifiche funzionali e dei casi d’uso;
            \item Progettazione dell’architettura tecnica della soluzione;
            \item Definizione dei flussi decisionali per la gestione delle anomalie;
        \end{itemize}
        \item \textbf{Quarta Settimana (40 ore) --- Realizzazione: Integrazione Elastic-AppDynamics}
        \begin{itemize}
            \item Implementazione dell’automazione Python per l’importazione dati da AppDynamics;
            \item Configurazione degli indici Elastic per i dati di traffico utente;
        \end{itemize}
        \item \textbf{Quinta Settimana (40 ore) --- Realizzazione: Sviluppo AI Agent core}
        \begin{itemize}
            \item Sviluppo del nucleo dell’AI Agent basato su tecnologia Elastic;
            \item Integrazione con il modello LLM scelto per l’analisi delle metriche;
            \item Implementazione del rilevamento anomalie di sicurezza e performance;
        \end{itemize}
        \item \textbf{Sesta Settimana (40 ore) --- Realizzazione: Dashboard e workflow}
        \begin{itemize}
            \item Realizzazione della dashboard Kibana per la visualizzazione delle anomalie;
            \item Implementazione del workflow di gestione automatizzata degli incidenti;
            \item Inizio stesura documentazione tecnica;
        \end{itemize}
        \item \textbf{Settima Settimana (40 ore) --- Realizzazione: Test e integrazione}
        \begin{itemize}
            \item Test del sistema end-to-end e validazione dei risultati;
            \item Ottimizzazione delle performance e correzione dei difetti rilevati;
            \item Proseguimento stesura documentazione;
        \end{itemize}
        \item \textbf{Ottava Settimana (40 ore) --- Documentazione: Stesura e revisione finale}
        \begin{itemize}
            \item Completamento della documentazione tecnica (architettura, flussi, esempi d’uso);
            \item Stesura della documentazione funzionale (use case);
            \item Revisione finale e consegna dei prodotti attesi;
        \end{itemize}
    \end{itemize}
}
